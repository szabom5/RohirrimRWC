111         public void actionPerformed(java.awt.event.ActionEvent event) {
112 
113             java.util.Set<org.jxmapviewer.viewer.Waypoint> waypoints

116             if (scan != null) {
117 
118                 try {
119 
120                     int time = 0, size = 0, minutes = 0;
121 
122                     time = scan.nextInt();
123                     minutes = scan.nextInt();
124                     size = scan.nextInt();
125 
126                     long ref_from = 0, ref_to = 0;
127                     int step = 0, maxstep = 1, type = 0;
128                     double lat, lon;
129                     double lat2, lon2;
130                     int num_captured_gangsters;
131                     String name = "Cop";
132 
133                     java.util.Map<String, Integer> cops = new java.util.HashMap<String, Integer>();
134 
135                     for (int i = 0; i < size; ++i) {
136 
137                         ref_from = scan.nextLong();
138                         ref_to = scan.nextLong();
139                         maxstep = scan.nextInt();
140                         step = scan.nextInt();
141                         type = scan.nextInt();
142 
143                         if (type == 1) {
144                             num_captured_gangsters = scan.nextInt();
145                             name = scan.next();
146 
147                             if (cops.containsKey(name)) {
148                                 cops.put(name, cops.get(name) + num_captured_gangsters);

150                                 cops.put(name, num_captured_gangsters);
151                             }
152                         }
153 
154                         lat = lmap.get(ref_from).lat;
155                         lon = lmap.get(ref_from).lon;
156 
157                         lat2 = lmap.get(ref_to).lat;
158                         lon2 = lmap.get(ref_to).lon;
159 
160                         if (maxstep == 0) {
161                             maxstep = 1;
162                         }
163 
164                         lat += step * ((lat2 - lat) / maxstep);
165                         lon += step * ((lon2 - lon) / maxstep);
166 
167                         if (type == 1) {
168                             waypoints.add(new WaypointPolice(lat, lon, name));
169                         } else if (type == 2) {
170                             waypoints.add(new WaypointGangster(lat, lon));
171                         } else if (type == 3) {
172                             waypoints.add(new WaypointCaught(lat, lon));
173                         } else {
174                             waypoints.add(new org.jxmapviewer.viewer.DefaultWaypoint(lat, lon));
175                         }
176 
177                     }
178 
179                     if (time >= minutes * 60 * 1000 / 200) {
180                         scan = null;
181                     }
182 
183                     StringBuilder sb = new StringBuilder();
184 
185                     int sec = time / 5;
186                     int min = sec / 60;
187                     sec = sec - min * 60;
188                     time = time - min * 60 * 5 - sec * 5;

191                     sb.append(min);

358     public static void readMap(java.util.Map<Long, Loc> lmap, String name) {
359 
360         java.util.Scanner scan;
361         java.io.File file = new java.io.File(name);
362 
363         long ref = 0;
364         double lat = 0.0, lon = 0.0;
365         try {
366 
367             scan = new java.util.Scanner(file);
368 
369             while (scan.hasNext()) {
370 
371                 ref = scan.nextLong();
372                 lat = scan.nextDouble();
373                 lon = scan.nextDouble();
374 
375                 lmap.put(ref, new Loc(lat, lon));
376 
377             }
378 
379         } catch (Exception e) {
380 
381             java.util.logging.Logger.getLogger(
382                     CarWindow.class
383                     .getName()).log(java.util.logging.Level.SEVERE, "hibás noderef2GPS leképezés", e);
384 
385         }
